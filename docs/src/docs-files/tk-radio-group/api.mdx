import { TkBadge } from "@takeoff-ui/react"; 

### Props

| Name | Type | Default | Description |
| ---- | ---- | ------- | ----------- |
| <TkBadge label="direction" variant="primary" size="large" type="filledlight"/> | <code>"horizontal", "vertical"</code> | 'horizontal' | The direction of the radio buttons. |
| <TkBadge label="invalid" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | false | Indicates whether the input is in an invalid state |
| <TkBadge label="label" variant="primary" size="large" type="filledlight"/> | <code>string</code> | null | Defines the label for the element. |
| <TkBadge label="position" variant="primary" size="large" type="filledlight"/> | <code>"left", "right"</code> | 'left' | Determines the position of the radio group and label. |
| <TkBadge label="showAsterisk" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | false | Displays a red asterisk (*) next to the label for visual emphasis. |
| <TkBadge label="spread" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | false | Determines whether the radios will spread evenly across the horizontal space. |
| <TkBadge label="type" variant="primary" size="large" type="filledlight"/> | <code>"card", "default"</code> | 'default' | Determines the appearance types of radios. |
| <TkBadge label="value" variant="primary" size="large" type="filledlight"/> | <code>any</code> | null | The value of the input. |

### Events

| Name | Description |
| ---- | ----------- |
| tk-change | Emitted when the value has changed. |
