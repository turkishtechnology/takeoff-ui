import { TkBadge } from "@takeoff-ui/react"; 

### Props

| Name | Type | Default | Description |
| ---- | ---- | ------- | ----------- |
| <TkBadge label="direction" variant="primary" size="large" type="filledlight"/> | <code>"horizontal", "vertical"</code> | 'horizontal' | The direction of the toggle button group. |
| <TkBadge label="rounded" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | false | The value of the rounded toggle button group. |
| <TkBadge label="type" variant="primary" size="large" type="filledlight"/> | <code>"basic", "divided", "light"</code> | 'basic' |  |
| <TkBadge label="value" variant="primary" size="large" type="filledlight"/> | <code>any</code> | null | The value of the selected toggle button. |

### Events

| Name | Description |
| ---- | ----------- |
| tk-change | Emitted when the selected value changes. |
