import { TkBadge } from "@takeoff-ui/react"; 

### Props

| Name | Type | Default | Description |
| ---- | ---- | ------- | ----------- |
| <TkBadge label="disabled" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | null | Disables the button, preventing user interaction. |
| <TkBadge label="icon" variant="primary" size="large" type="filledlight"/> | <code>IIconOptions, string</code> | null | Specifies a material icon name to be displayed. |
| <TkBadge label="iconPosition" variant="primary" size="large" type="filledlight"/> | <code>"left", "right"</code> | 'left' | Defines the position of the icon. |
| <TkBadge label="label" variant="primary" size="large" type="filledlight"/> | <code>string</code> | '' | Label text displayed inside the button. |
| <TkBadge label="rounded" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | null | Makes the button round with an icon-only style. |
| <TkBadge label="selected" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | false | Whether the button is selected. |
| <TkBadge label="size" variant="primary" size="large" type="filledlight"/> | <code>"base", "large", "small"</code> | 'base' | Sets size for the component. |
| <TkBadge label="type" variant="primary" size="large" type="filledlight"/> | <code>"filled", "filled-light", "outlined", "raised", "text"</code> | 'filled' | The value of the type toggle button. |
| <TkBadge label="value" variant="primary" size="large" type="filledlight"/> | <code>any</code> | null | The value of the toggle button. |
| <TkBadge label="variant" variant="primary" size="large" type="filledlight"/> | <code>"neutral", "primary"</code> | 'neutral' | Determines the button's variant for different styles. |

### Events

| Name | Description |
| ---- | ----------- |
| tk-toggle | Emitted when the toggle button is toggled. |
