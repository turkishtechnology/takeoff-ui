import { TkBadge } from "@takeoff-ui/react"; 

### Props

| Name | Type | Default | Description |
| ---- | ---- | ------- | ----------- |
| <TkBadge label="label" variant="primary" size="large" type="filledlight"/> | <code>string</code> | null | Sets the label of the spinner component. |
| <TkBadge label="orientation" variant="primary" size="large" type="filledlight"/> | <code>"horizontal", "vertical"</code> | 'vertical' | Controls the orientation of the spinner component. |
| <TkBadge label="size" variant="primary" size="large" type="filledlight"/> | <code>"base", "large", "small", "xlarge", "xsmall", "xxsmall"</code> | 'base' | Controls the size of the spinner component. |
| <TkBadge label="type" variant="primary" size="large" type="filledlight"/> | <code>"dots", "lines", "loader", "pulse", "rounded", "three-dots"</code> | 'rounded' | Sets the style of spinner component. |
