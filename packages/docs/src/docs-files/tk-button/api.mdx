import { TkBadge } from "@takeoff-ui/react"; 

### Props

| Name | Type | Default | Description |
| ---- | ---- | ------- | ----------- |
| <TkBadge label="disabled" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | null | Disables the button, preventing user interaction. |
| <TkBadge label="fullWidth" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | null | Sets the button to full-width mode, making it span the container. |
| <TkBadge label="href" variant="primary" size="large" type="filledlight"/> | <code>string</code> | null | Sets the URL the button should navigate to when clicked (for type="link" buttons). |
| <TkBadge label="icon" variant="primary" size="large" type="filledlight"/> | <code>IIconOptions, string</code> | null | Specifies a material icon name to be displayed. |
| <TkBadge label="iconPosition" variant="primary" size="large" type="filledlight"/> | <code>"left", "right"</code> | 'left' | Defines the position of the icon. |
| <TkBadge label="label" variant="primary" size="large" type="filledlight"/> | <code>string</code> | '' | Label text displayed inside the button. |
| <TkBadge label="loading" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | null | Shows a loading icon inside the button. |
| <TkBadge label="mode" variant="primary" size="large" type="filledlight"/> | <code>"button", "link", "reset", "submit"</code> | 'button' | Sets the button type. |
| <TkBadge label="rounded" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | null | Makes the button round with an icon-only style. |
| <TkBadge label="size" variant="primary" size="large" type="filledlight"/> | <code>"base", "large", "small"</code> | 'base' | Sets size for the component. |
| <TkBadge label="target" variant="primary" size="large" type="filledlight"/> | <code>string</code> | null | Specifies where to open the linked document (for type="link" buttons). |
| <TkBadge label="type" variant="primary" size="large" type="filledlight"/> | <code>"elevated", "filled", "outlined", "text"</code> | 'filled' | This field specifies the design type of the component. |
| <TkBadge label="underline" variant="primary" size="large" type="filledlight"/> | <code>boolean</code> | null | Applies underline styling to the button (for type="link" buttons). |
| <TkBadge label="variant" variant="primary" size="large" type="filledlight"/> | <code>"danger", "info", "neutral", "primary", "secondary", "success", "warning", "white"</code> | 'primary' | Determines the button's variant for different styles. |

### Events

| Name | Description |
| ---- | ----------- |
| tk-click | Emitted when the button click. |
